This was posted by Wayne D. Hoxsie Jr. <hoxsiew@crl.com>:

#include <stdio.h>

v(char *s)
{
  int i=0,j=0,k;

  k=!(!s[16]);
  for(;*s;*s++)
    i+=(++j%2-k)?(*s-'0')*2>9?((*s-'0')*2)-9:(*s-'0')*2:*s-'0';
  return (i%10);
}
main()
{
  char s[160];

  printf("Enter credit card number\n");
  scanf("%s",s);
  printf("Credit card number is %svalid\n",v(s)?"in":"");
}
